PROMPT_PATH = "prompt.txt"

def encode_image(image_path):
    with open(image_path, "rb") as img_file:
        return base64.b64encode(img_file.read()).decode("utf-8")

def load_prompt():
    with open(PROMPT_PATH, "r", encoding="utf-8") as f:
        return f.read()

    base64_image = encode_image(image_path)
    
    response = openai.ChatCompletion.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": prompt},
            {"role": "user", "content": [

                {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{base64_image}"}}
            ]}
        ],
        max_tokens=1800
    )
    return response.choices[0].message.content

def save_output(filename, content):
    with open(os.path.join(OUTPUT_PATH, filename + ".txt"), "w", encoding="utf-8") as f:
        f.write(content)

    prompt = load_prompt()
    os.makedirs(OUTPUT_PATH, exist_ok=True)

    for file in os.listdir(IMAGES_PATH):

            save_output(file.split('.')[0], result)
            print(f"âœ… Resultado guardado para {file}")

if __name__ == "__main__":
    main()

import streamlit as st

from PIL import Image
import base64
from config import GEMINI_API_KEY

# Cargar prompt OSINT
def load_prompt():
    with open("prompt.txt", "r", encoding="utf-8") as f:
        return f.read()

# Codificar imagen para Gemini
def image_to_bytes(img):
    buffered = img.convert("RGB")
    return buffered

# Procesar imagen con Gemini
def analyze_with_gemini(img, prompt):
    model = genai.GenerativeModel("gemini-pro-vision")

    response = model.generate_content(
        [prompt, img],
        generation_config={"temperature": 0.6, "max_output_tokens": 2048}
    )
    return response.text

# Streamlit UI

st.title("ğŸŒ GeoIntel OSINT con Gemini")
st.markdown("Sube una imagen y descubre dÃ³nde fue tomada con IA gratuita de Google.")

    image = Image.open(uploaded_file)
    st.image(image, caption="Imagen cargada", use_column_width=True)

    if st.button("ğŸ” Analizar"):
        st.spinner("Analizando imagen con Gemini...")

        img_bytes = image_to_bytes(image)
        prompt = load_prompt()
        result = analyze_with_gemini(img_bytes, prompt)

        st.success("âœ… AnÃ¡lisis completado")
        st.markdown("### ğŸ§­ Resultado:")
        st.markdown(result)

        # Extraer coordenadas y mostrar mapa (opcional)

        coords = re.search(r"\[(\-?\d+\.\d+),\s*(\-?\d+\.\d+)\]", result)

            sv_url = f"https://www.google.com/maps/@?api=1&map_action=pano&viewpoint={lat},{lon}"
            st.markdown(f"ğŸ“ [Ver en Google Maps]({maps_url})")
            st.markdown(f"ğŸ“¸ [Ver en Street View]({sv_url})")
